def intl_format(n):
    return "{:,.2f}".format(n)

def nepali_format(n):
    s="{:.2f}".format(n)
    parts=s.split(".")
    x=parts[0]
    r=""
    if len(x)>3:
        r=",".join([x[:-3][::-1][i:i+2] for i in range(0,len(x)-3,2)])[::-1] + "," + x[-3:]
    else:
        r=x
    return r + "." + parts[1]

num=111625.92822
print(nepali_format(num))
print(intl_format(num))
